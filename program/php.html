    # 分配

    :- -
    a += b 如同 a = a + b
    a -= b 如同 a = a – b
    a *= b 如同 a = a * b
    a /= b 如同 a = a / b
    a %= b 如同 a = a % b

    # 比较

    :- -
    == 平等的
    === 完全相同的
    != 不相等
    <> 不相等
    !== 不相同
    < 少于
    > 比...更棒
    <= 小于或等于
    >= 大于或等于
    <=> 小于/等于/大于

    # 逻辑的

    :- -
    and 和
    or 或者
    xor 独家或
    ! 不是
    && 和
    \|\| 或者
