    window.addEventListener('DOMContentLoaded', () => {
  const replaceText = (selector, text) => {
    const element = document.getElementById(selector)
    if (element) element.innerText = text
  }

  for (const dependency of ['chrome', 'node', 'electron']) {
    replaceText(`${dependency}-version`, process.versions[dependency])
  }
})

    # 启动时激活主实例窗口的示例

    const { app } = require('electron');
let myWindow = null;

const additionalData = { myKey: 'myValue' };
const gotTheLock = app.requestSingleInstanceLock(additionalData);

if (!gotTheLock) {
  app.quit();
} else {
  app.on(
    'second-instance',
    (event, commandLine, workingDirectory, additionalData) => {
      // 输出从第二个实例中接收到的数据
      console.log(additionalData);

      // 有人试图运行第二个实例，我们应该关注我们的窗口
      if (myWindow) {
        if (myWindow.isMinimized()) myWindow.restore();
        myWindow.focus();
      }
    }
  )

  // 创建 myWindow, 加载应用的其余部分, etc...
  app.whenReady().then(() => {
    myWindow = createWindow();
  })
}

    # 事件

    :- :-
    will-finish-launching 基础的启动的时候被触发 # (opens new window)
    ready 完成初始化时，触发一次 # (opens new window)
    window-all-closed 在程序关闭窗口前发信号 # (opens new window)
    before-quit 当所有窗口被关闭后触发，应用程序将退出 # (opens new window)
    will-quit 所有窗口被关闭后触发，应用程序将退出 # (opens new window)
    quit 在应用程序退出时触发 # (opens new window)
    open-file (mac) 在应用中打开一个文件时触发 # (opens new window)
    open-url (mac) 应用中打开一个 URL 时触发 # (opens new window)
    activate (mac) 当应用被激活时触发 # (opens new window)
    did-become-active (mac) 不同设备的活动想要恢复时在切换期间触发 # (opens new window)
    continue-activity (mac) 不同设备的活动通过 Handoff 想要恢复时触发 # (opens new window)
    will-continue-activity (mac) 恢复来自不同设备的活动之前在切换期间触发 # (opens new window)
    continue-activity-error (mac) 不同设备的活动无法恢复时在切换期间触发 # (opens new window)
    activity-was-continued (mac) 活动在另一个设备上成功恢复后切换期间触发 # (opens new window)
    update-activity-state (mac) 当即将在另一台设备上恢复切换时触发 # (opens new window)
    new-window-for-tab (mac) 用户点击原生的 macOS 新标签按钮时触发 # (opens new window)
    browser-window-blur 当 browserWindow 变得模糊时触发 # (opens new window)
    browser-window-focus 当 browserWindow 获得焦点时触发 # (opens new window)
    browser-window-created 创建新的 browserWindow 时触发 # (opens new window)
    web-contents-created 创建新的 webContents 时触发 # (opens new window)
    certificate-error 当对 url 的 certificate 证书验证失败触发 # (opens new window)
    select-client-certificate 当一个客户证书被请求的时候发出 # (opens new window)
    login 当 webContents 要进行基本身份验证时触发 # (opens new window)
    gpu-info-update 每当有 GPU 信息更新时触发 # (opens new window)
    gpu-process-crashed 废弃 当 gpu 进程崩溃或关闭(杀死)时触发 # (opens new window)
    renderer-process-crashed 废弃 渲染器进程崩溃或关闭(杀死)时触发 # (opens new window)
    render-process-gone 渲染器进程意外消失时触发 # (opens new window)
    child-process-gone 子进程意外消失时触发 # (opens new window)
    accessibility-support-changed 当 Chrome 的辅助功能状态改变时触发 # (opens new window)
    session-created 创建了一个新的 session 后被触发 # (opens new window)
    second-instance 在你的应用程序的首个实例中触发 # (opens new window)

    # BrowserWindow

    # 参数

    :- :-
    width 整数 窗口的宽度(以像素为单位)。默认值 (800)
    height 整数 窗口的高度(以像素为单位)。默认值 (600)
    x 整数 (可选) 窗口相对于屏幕左侧偏移量。默认值窗口居中
    y 整数 (可选) 窗口相对于屏幕顶端偏移量。默认值窗口居中
    useContentSize boolean (宽) 和 (高) 设置为web页面的尺寸。默认值 (false)
    center boolean 窗口是否在屏幕居中。默认值 (false)
    minHeight 整数 窗口的最小高度。默认值 (0)
    maxWidth 整数 窗口的最大宽度。默认值不限
    maxHeight 整数 窗口的最大高度。默认值不限
    resizable boolean 窗口大小是否可调整。默认值 (true)
    movable boolean (win/mac) 窗口是否可移动。默认值 (true)
    minimizable boolean (win/mac) 窗口是否可最小化。默认值 (true)
    maximizable boolean (win/mac) 窗口是否最大化。默认值 (true)
    closable boolean (win/mac) 窗口是否可关闭。默认值 (true)
    focusable boolean 窗口是否可以聚焦. 默认值 (true)。在 Windows 中设置 (focusable: false) 也意味着设置了 (skipTaskbar: true) 在 Linux 中设置 (focusable: false) 时窗口停止与 wm 交互, 并且窗口将始终置顶
    alwaysOnTop boolean 窗口是否永远在别的窗口的上面。默认值 (false)
