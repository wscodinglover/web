# run/exec 运行命令

# 在所有包含它的包中运行 npm run my-script
$ lerna run <script> -- [..args]
$ lerna run test
$ lerna run build

# 观看所有包并在更改时转换，流式前缀输出
$ lerna run --parallel watch

# 在所有包中运行命令
$ lerna exec -- <command> [..args]
$ lerna exec -- rm -rf ./node_modules
$ lerna exec -- protractor conf.js

--npm-client <client> # (opens new window) 必须是知道如何运行 npm 生命周期脚本的可执行文件，默认值是 npm
--stream # (opens new window) 立即从子进程流式输出，以原始包名称为前缀
--parallel # (opens new window) 类似于 --stream 但完全忽略并发和拓扑排序，立即在所有匹配的包中运行给定的命令或脚本，并带有前缀流输出
--no-bail # (opens new window) 默认情况下，如果任何脚本运行返回非零退出代码，lerna run 将退出并出现错误。传递 --no-bail 以禁用此行为
--no-prefix # (opens new window) 当输出为流式传输（--stream 或 --parallel）时禁用包名称前缀。当将结果传送到其他进程（例如编辑器插件）时，此选项很有用
--profile # (opens new window) 分析脚本执行并生成性能配置文件
--profile-location <location> # (opens new window) 您可以为性能配置文件输出提供自定义位置。提供的路径将相对于当前工作目录进行解析。
useNx=false # (opens new window) 通过将 useNx 设置为 false，您可以使用 lerna 中的遗留任务运行实现 (p-map 和 p-queue)，而不是使用由 Nx (opens new window) 提供支持的默认现代任务运行器实现。

# publish 发布包

# 发布自上次发布以来已更改的软件包
$ lerna publish              
# 显式发布在当前提交中标记的包
$ lerna publish from-git     
# 显式发布注册表中不存在最新版本的软件包
$ lerna publish from-package
# 使用下一个语义预发布版本，例如
$ lerna publish --canary
# 1.0.0 => 1.0.1-alpha.0+${SHA}
# 自上次提交以来更改的包
# 随后的金丝雀发布将产生1.0.1-alpha.1+${SHA}等

$ lerna publish --canary --preid beta
# 1.0.0 => 1.0.1-beta.0+${SHA}

# 以下是等价的：
$ lerna publish --canary minor
$ lerna publish --canary preminor
# 1.0.0 => 1.1.0-alpha.0+${SHA}

:- :-
--canary # (opens new window) 使用此标志运行时，以更精细的方式（每次提交）发布包
--contents <dir> # (opens new window) 要发布的子目录。 必须适用于所有包，并且必须包含 package.json 文件
--dist-tag <tag> # (opens new window) 使用此标志运行时，将使用给定的 npm dist-tag（默认为 latest）发布到 npm
--git-head <sha> # (opens new window) 打包 tarball 时将显式 SHA 设置为清单上的 gitHead，仅允许使用 from-package 位置
--graph-type <all\|dependencies> # (opens new window) 设置在构建包图时使用哪种依赖项。默认值是依赖项，即仅包含包的 package.json 的依赖项部分中列出的包
--ignore-scripts # (opens new window) 传递时，此标志将在 lerna 发布期间禁用运行生命周期脚本 (opens new window)
--ignore-prepublish # (opens new window) 传递时，此标志将禁用在 lerna 发布期间运行已弃用 (opens new window)的预发布脚本 (opens new window)
--legacy-auth # (opens new window) 发布需要身份验证的包时，您正在使用仅使用旧版 Base64 用户名:密码 的内部托管 NPM 注册表。这与 NPM 发布 _auth 标志相同
--no-git-reset # (opens new window) 默认情况下，lerna publish 确保对工作树的任何更改都已重置
--no-granular-pathspec # (opens new window) 默认情况下，lerna publish 将尝试（如果启用）git checkout 仅在发布过程中临时修改的叶包清单
--verify-access # (opens new window) 从历史上看，lerna 试图通过使用给定令牌执行一些抢占式 npm API 请求来快速解决授权/身份验证问题
--otp # (opens new window) 发布需要双重身份验证的包时，您可以使用 --otp 指定一次性密码
--preid # (opens new window) 与同名的 lerna 版本选项不同，该选项只适用于 --canary 版本计算
--pre-dist-tag <tag> # (opens new window) 与 --dist-tag 的工作方式相同，但仅适用于使用预发布版本发布的软件包
--registry <url> # (opens new window) 使用此标志运行时，转发的 npm 命令将为您的包使用指定的注册表
--tag-version-prefix # (opens new window) 此选项允许提供自定义前缀而不是默认前缀：v
--temp-tag # (opens new window) 传递时，此标志将更改默认发布过程，首先将所有更改的包发布到临时 dist-tag(lerna-temp)，然后将新版本移动到 --dist-tag 配置的 dist-tag (默认latest)
--yes # (opens new window) 使用此标志运行时，lerna publish 将跳过所有确认提示

--skip-npm # (opens new window) 直接调用 lerna version

# version 修改版本号

$ lerna version 1.0.1 # 明确的
$ lerna version patch # semver 关键字
$ lerna version       # 从提示中选择

$ lerna version [major | minor | ...]
# 使用下一个语义版本值
# 这会跳过“为...选择新版本”提示
# 强制所有包版本化
$ lerna version --force-publish
$ lerna version -m "chore(doc): publish %s"
# 提交消息 = "chore(doc): publish v1.0.0"
$ lerna version -m "chore(doc): publish %v"
# 提交消息 = "chore(doc): publish 1.0.0"

--conventional-commits # (opens new window) 使用常规提交规范来确定版本 bump (opens new window) 并生成 CHANGELOG.md 文件

--conventional-prerelease # (opens new window) 预发布版本发布指定的包
--create-release <type> # (opens new window) 根据更改的包创建正式的 GitHub 或 GitLab 版本

--ignore-scripts # (opens new window) 禁用在 lerna version 期间运行的生命周期脚本
--include-merged-tags # (opens new window) 在检测到更改的包时包括来自合并分支的标签
--message <msg> # (opens new window) 此选项别名为 -m 以与 git commit 进行奇偶校验
--no-changelog # (opens new window) 使用常规提交时，不要生成任何 CHANGELOG.md 文件
--no-commit-hooks # (opens new window) 允许 git commit hooks 在提交版本更改时运行。通过 --no-commit-hooks 禁用此行为

--no-push # (opens new window) 将已提交和标记的更改推送到配置的 git remote。通过 --no-push 禁用此行为
--preid # (opens new window) 使用此标志运行时，lerna 版本将使用指定的预发布标识符 (opens new window)增加 premajor、preminor、prepatch 或 prerelease semver bumps
--sign-git-commit # (opens new window) 此选项类似于同名的 npm 版本选项
--sign-git-tag # (opens new window) 此选项类似于同名的 npm 版本选项
--force-git-tag # (opens new window) 此选项替换任何现有标记而不是失败
--tag-version-prefix # (opens new window) 此选项允许提供自定义前缀而不是默认前缀：v
--yes # (opens new window) 使用此标志运行时，lerna 版本将跳过所有确认提示

# 选项

:- :-
--scope <glob> # (opens new window) 仅包括名称与给定 glob 匹配的包
--ignore <glob> # (opens new window) 排除名称与给定 glob 匹配的包
--no-private # (opens new window) 排除私有包
--since [ref] # (opens new window) 仅包括自指定 ref 以来已更改的包
--exclude-dependents # (opens new window) 使用 --since 运行命令时排除所有传递依赖项，覆盖默认的“changed”算法

# add

将依赖项添加到匹配的包

$ lerna add <package>[@version] \
        [--dev] [--exact] [--peer]
1
2

选项

:- :-
--dev # (opens new window) 将新包添加到 devDependencies
--exact # (opens new window) 添加具有精确版本（例如 1.0.1）而不是默认 ^ semver 范围（例如 ^1.0.1）的新包
--peer # (opens new window) 将新包添加到 peerDependencies
--registry <url> # (opens new window) 使用自定义注册表安装目标包
--no-bootstrap # (opens new window) 跳过链式 lerna bootstrap

实例

# 将 mod-1 包添加到“prefix-”前缀文件夹中的包中
$ lerna add mod-1 packages/prefix-*
# 将 mod-1 安装到mod-2
$ lerna add mod-1 --scope=mod-2
# 在 devDependencies 中安装 mod-1 到 mod-2
$ lerna add mod-1 --scope=mod-2 --dev
# 在 peerDependencies 中安装 mod-1 到 mod-2
$ lerna add mod-1 --scope=mod-2 --peer
# 在除 mod-1 之外的所有模块中安装 mod-1
$ lerna add mod-1
# 在所有模块中安装 babel-core
$ lerna add babel-core
